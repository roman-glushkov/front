# Разбор ProjectTitle.tsx

Этот компонент отображает заголовок презентации, поле для его редактирования и информацию о количестве слайдов и выбранном слайде. Используется в верхней части интерфейса.

Команды и конструкции

import React from 'react' — импорт React для работы с JSX.
import { Presentation } from '../store/types/presentation' — импорт типа Presentation, описывающего структуру объекта презентации.
import './ProjectTitle.css' — подключение CSS-стилей компонента.

interface Props { ... } — описание типа пропсов, которые принимает компонент:
pres — объект презентации с заголовком и слайдами.
onTitleChange — функция-обработчик изменения текста в поле ввода.
onTitleCommit — обработчик, вызывающийся при потере фокуса (подтверждение нового названия).
onTitleKeyDown — обработчик нажатий клавиш в поле ввода (например, Enter).
selSlideId — id выбранного слайда.

export default function ProjectTitle({ ... }: Props) — определение функционального компонента с деструктуризацией пропсов.

return (...) — JSX-разметка компонента.

<div className="presentation-info top"> — контейнер с информацией о презентации.  
<h3>Презентация: {pres.title}</h3> — отображает текущий заголовок презентации.  
<input value={pres.title} onChange={onTitleChange} onBlur={onTitleCommit} onKeyDown={onTitleKeyDown} /> — поле для редактирования названия:
value — текущее значение заголовка,  
onChange — обновление текста при вводе,  
onBlur — сохранение при уходе фокуса,  
onKeyDown — проверка нажатий клавиш (Enter).  
<p>Слайдов: {pres.slides.length} | Выбран: {selSlideId || 'нет'}</p> — отображает количество слайдов и id выбранного слайда (если ничего не выбрано — выводит «нет»).

Компонент не имеет собственного состояния — всё управление передано из родительского компонента через пропсы.
