# Разбор App.tsx

Это главный компонент приложения презентаций. Он управляет состоянием презентации, слайдов и элементов, отображает интерфейс (слайды, рабочую область, панель инструментов) и обрабатывает действия пользователя (добавление/удаление слайдов, элементов, изменение текста, позиции и размера).

Команды и конструкции

React — локальная переменная, получает default export из react  
useState — хук React для состояния в функциональном компоненте  
function App() — определение функционального компонента App

[pres, setPres] = useState(initialPresentation) — состояние презентации  
[selSlideId, setSelSlideId] = useState('') — состояние выбранного слайда  
[selElId, setSelElId] = useState('') — состояние выбранного элемента

updateSlide - функция обновления конкретного слайда через функцию updater  
pres.slides.find(...) — поиск слайда по id  
.map(...) — перебор массива
prev - предыдущее значение
...prev - копирование предыдущего значения
slides - заменяем только список слайдов.
prev.slides.map((s) => (s.id === selSlideId ? updater(s) : s)) - смотрим все слайды, если слайд тот, то обновляем слайд, если нет оставляем все как есть

switch (action) { ... } — обработка действий пользователя  
case '...' — конкретное действие, например 'Добавить слайд'

handleTitleChange = (e) => { ... } — обработчик изменения текста в поле ввода.
e — объект события изменения (React.ChangeEvent<HTMLInputElement>).
e.target.value — текущее значение поля input.
handleTitleKeyDown = (e) => { ... } — обработчик нажатий клавиш в поле ввода.
e.key === 'Enter' — проверка, нажал ли пользователь Enter.
e.currentTarget.blur() — снимает фокус с input (как будто пользователь кликнул вне поля), чтобы завершить ввод.
handleTitleCommit = (e) => { ... } — обработчик завершения ввода названия презентации.
handleSlideClick = (slideId, index) => { ... } — обработчик клика по слайду  
handleElementClick = (elementId, slideId) => { ... } — обработчик клика по элементу слайда

return (...) — JSX-дерево компонента
value={pres.title} — значение инпута
onChange, onBlur, onKeyDown — обработчики событий (реагируют на ввод, уход фокуса и Enter).

.map(...) — перебор массива и генерация элементов JSX  
? : — тернарный оператор  
key={el.id} — уникальный ключ для списка элементов  
style={{ ... }} — inline-стили элемента  
export default App — экспорт компонента по умолчанию
