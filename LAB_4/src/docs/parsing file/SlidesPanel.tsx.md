# Разбор SlidesPanel.tsx

Этот компонент отвечает за отображение панели со списком слайдов. Он показывает миниатюры (упрощённые представления) всех слайдов и позволяет выбрать нужный при клике.

Команды и конструкции

import React from 'react' — импорт React для работы с JSX.
import { Slide } from '../store/types/presentation' — импорт типа Slide, описывающего структуру одного слайда.
import './SlidesPanel.css' — подключение CSS-стилей для панели слайдов.

interface Props { ... } — описание пропсов, которые принимает компонент:
slides — массив слайдов презентации.
selectedSlideId — id выбранного в данный момент слайда.
onSlideClick — функция-обработчик клика по слайду, получает id слайда и его порядковый номер.

export default function SlidesPanel({ slides, selectedSlideId, onSlideClick }: Props) — определение функционального компонента с деструктуризацией пропсов.

return (...) — JSX-разметка панели слайдов.

<div className="slides-panel"> — контейнер панели.  
<h3>Слайды</h3> — заголовок панели.  
<div className="slides-container"> — область, в которой размещены все слайды.

slides.map((s, i) => ( ... )) — перебор массива слайдов и создание элементов для каждого.

<div key={s.id} ...> — блок для отдельного слайда,  
key={s.id} — уникальный ключ для корректного рендеринга,  
onClick={() => onSlideClick(s.id, i)} — обработчик клика, передает id и индекс слайда,  
className={`simple-slide ${selectedSlideId === s.id ? 'selected' : ''}`} — применение класса выделения, если слайд выбран.  
Текст внутри — «Слайд {i + 1}», то есть порядковый номер слайда.

Компонент не хранит состояние — он просто отображает список и сообщает родительскому компоненту, какой слайд выбрал пользователь.
